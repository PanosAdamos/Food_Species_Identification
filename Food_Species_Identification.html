# Build a mini fastQC

def gc_content(sequence):
    seq = sequence.upper()
    gc_count = seq.count("G") + seq.count("C")
    gc_percentage = (gc_count/len(seq))*100
    return gc_percentage 

# Calculate the gc content in all reads

gc_values = [gc_content(rec.seq) for rec in records]
